

az account set -s "Microsoft Azure"

AKS Requires
Microsoft Container Service
Microsoft.compute
Microsoft.Network
Microsoft.Storage

az provider register -n Microsoft.compute

az provider list -o table

create resource aks to use
az group create -n kubeclust-aks --location eastus

get aks supported version number
az aks get-versions --location eastus --output table

create cluster in
az aks create --resource-group kubeclust1-aks --name kubeclust-aks --node-count 1 --generate-ssh-keys --kubernetes-version 1.9.10 --node-vm-size=Standard_B2s

az aks get-credentials --resource-group kubeclust-aks --name kubeclust-aks

kubectl get nodes

kubectl cluster-info
